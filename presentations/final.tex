\frametitle{Results}
\begin{itemize}
\item<1-> 12 people completed the game and post-test
\item<1-> Pre-test mean: 3.167 (s = 1.267)
\item<1-> Post-test mean: 4.25 (s = 0.754)
\item<2-> Statistically significant
    \begin{itemize}
    \item paired t-test
    \item t(11) = 3.767, p $\approx$ 0.003
    \end{itemize}
\end{itemize}
